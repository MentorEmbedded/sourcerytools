#include <vsip/initfin.hpp>

int main(void)
{
  vsip::vsipl prog();

  const vsip::Matrix<vsip::scalar_i> constM(2,2,99);
  vsip::scalar_i a = constM(0,0); // fails to compile

  vsip::Matrix<vsip::scalar_i> m(2,2);
  m = vsip::scalar_i(99); // fails to compile

  return 0;
}
